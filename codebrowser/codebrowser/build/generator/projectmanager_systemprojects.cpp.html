22#include "projectmanager.h"
23
24#include <sstream>
25
26std::vector<ProjectInfo> ProjectManager::systemProjects()
27{
28 std::vector<ProjectInfo> projects;
29
30 std::istringstream stream("include;/usr/include/");
31 std::string current;
32 std::string name;
33
34 while (std::getline(stream, current, ';')) {
35 if(name.empty()) {
36 name = current;
37 } else {
38 projects.emplace_back(name, current, ProjectInfo::Internal);
39 name.clear();
40 }
41 }
42
43 return projects;
44}
